class Solution {
public:
    int t[46];
  int solve(int n){    
    if(n<3){
     return t[n]=n;
    }
     if(t[n]!=-1){
         return t[n];
     } 
    return t[n]=solve(n-1)+solve(n-2);
    return t[n];  
  }

int climbStairs(int n) {
    memset(t,-1,sizeof(t));
    for(int i=0;i<46;i++){
        if(i<3){
            t[i]=i;
        }else{
            t[i]=t[i-1]+t[i-2];
        }
    }
    return t[n];
}
};
